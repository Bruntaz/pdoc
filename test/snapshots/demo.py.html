This is a demo module to show what's possible!

from dataclasses import dataclass, field
from typing import TypeVar

CONST_A: int = 42
"""A happy constant"""

T = TypeVar('T')


def foo(a: str, b: "Foo", *, c: T = False) -> T:
    """Do foo, but do it quickly."""
    return 42


CONST_B = "yes"
"""A non-annotated constant"""

CONST_C: float
"""An unset constant"""


class Foo:
    foo = "lots of foo!"
    '''The fooest of all attributes.'''

    def __init__(self):
        self.a: int = 42
        """I am defined in the constructor only"""

    def do_foo(self) -> "Foo":
        """You can actually do foo, too!"""
        return self

    async def i_am_async(self) -> int:
        pass


class Bar(Foo):
    bar: str
    """I sneaked in. :)"""

    class Baz:
        baz = True


def wat(x: Bar.Baz = "a test") -> int:
    return False


class DoubleInherit(Foo, Bar.Baz):
    """I am rich."""

class DataDemo:

    b: str
    """I am documented!"""
    c: bool = field(repr=False)
